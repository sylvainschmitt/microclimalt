  filter(type == "Fourier") %>% 
  fft_roll(time_window = 5*24,
           time_col = "time",
           temperature_col = "tas") %>%

  select(date, period, power) %>% 
  rename(power_era = power)

g <- read_tsv("data/derived_data/fc_all.tsv") %>% 
  mutate(date = as_date(datetime)) %>% 
  left_join(era_f) %>% 
  filter(period %in% c(0,12)) %>% 
  select(-coefficient, -frequency) %>% 
  na.omit() %>% 
  mutate(ratio = power/power_era) %>% 
  select(-power, -power_era) %>% 
  mutate(period = paste0("period_", period)) %>% 
  na.omit() %>% 
  group_by(id, type, period, hydric_year, season) %>% 
  summarise(q05 = quantile(ratio, .05),

palette <-  c(
  "Air"     = "#56B4E9",   # sky blue
  "Surface" = "#009E73",   # orange
  "Soil"    = "#E69F00"    # green
)
ratios <- read_tsv("data/derived_data/fc_all.tsv") %>% 
  mutate(date = as_date(datetime)) %>% 
  left_join(era_f) %>% 
  filter(!is.na(power_era)) %>% 
  filter(period %in% c(0, 24)) %>% 
  mutate(ratio = power/power_era) %>% 
  group_by(period, type, hydric_year, season, id) %>% 
  summarise(ratio = mean(ratio)) %>% 
  na.omit()
pad <- read_tsv("data/derived_data/sensors_pad_xy.tsv") %>% 
  rename(x = x_p, y = y_p) %>%
  left_join(read_tsv("data/derived_data/pad.tsv")) %>% 
  group_by(id, h) %>% 
  summarise(pad = median(pad)) %>% 
  mutate(strata = ifelse(h < 10, "understory", "upperstory")) %>% 
  group_by(id, strata) %>% 
  summarise(pad = sum(pad)) %>% 
  mutate(strata = paste0("pad_", strata)) %>% 
  pivot_wider(names_from = "strata", values_from = "pad")
topo <- read_tsv("data/derived_data/coords.tsv") %>% 
  mutate(twi = extract(rast("data/raw_data/topo/Paracou_ALT_TWI_5m.tif"),
                       read_tsv("data/derived_data/coords.tsv")[2:3])$Paracou_TWI_5m) %>% 
  select(id, twi)
model_all <- ratios %>% 
  left_join(pad) %>% 
  left_join(topo) %>% 
  mutate(season = factor(season, levels = c("wet", "dry"))) %>% 
  mutate(hydric_year = recode(hydric_year, 
                               "2024-2025" = "normal", 
                              "2023-2024" = "El Nino")) %>% 
  mutate(hydric_year = factor(hydric_year, levels = c("normal", "El Nino"))) %>% 
  ungroup() %>% 
  mutate(across(c("twi", "pad_understory", "pad_upperstory"), scale)) %>% 
  nest_by(period, type) %>% 
  mutate(
    fit = list(lm(ratio ~ season*hydric_year + twi + pad_understory + pad_upperstory, data = data)),
    glance = list(broom::glance(fit) %>% 
                    select(adj.r.squared, nobs)),
    estimates = list(broom::tidy(fit, conf.int = TRUE) %>% 
                       select(term, estimate, p.value, conf.low, conf.high))
  ) %>% 
  select(type, glance, estimates) %>% 
  unnest() %>% 
  filter(term != "(Intercept)") %>% 
  rename_all(~ gsub(".", "_", ., fixed = TRUE)) %>% 
  mutate(period = recode(as.character(period),
                         "0" = "Mean temperature ratio",
                         "24" = "24-h amplitude ratio")) %>% 
  mutate(period = factor(period, levels = c("Mean temperature ratio",
                                            "24-h amplitude ratio"))) %>% 
  mutate(type = factor(type, levels = c("air", "surface", "soil"))) %>% 
  mutate(term = recode(term,
                       "twi" = "TWI",
                       "seasondry" = "Dry season",
                       "hydric_yearEl Nino" = "El Niño year",
                       "pad_understory" = "PAD understory",
                       "pad_upperstory" = "PAD upperstory",
                       "seasondry:hydric_yearEl Nino" = "Dry season x El Niño year"
                       )) %>% 
  mutate(term = factor(term, levels = c("Dry season",
                                        "El Niño year",
                                        "Dry season x El Niño year",
                                        "TWI",
                                        "PAD understory",
                                        "PAD upperstory") %>% 
                         rev()))
rsquared <- select(model_all, adj_r_squared) %>% unique()
g <- model_all %>% 
  ggplot(aes(term, estimate, col = type, alpha = (p_value < 0.05))) +
  geom_hline(yintercept = 0, col = "darkgrey") +
  geom_segment(aes(y = conf_low, yend = conf_high), 
               position = position_dodge(width = 0.3, reverse = TRUE)) +
  geom_point(position = position_dodge(width = 0.3, reverse = TRUE)) +
  theme_bw() +
  xlab("") + ylab("") +
  scale_alpha_discrete(guide = "none", range = c(0.2, 1)) +
  coord_flip() +
  scale_color_discrete("", palette = palette) +
  theme(legend.position = "bottom") +
  facet_wrap(~ period) +
  theme(panel.grid = element_blank(),
        axis.text.x = element_text(angle = 0, size = 12),
        axis.text.y = element_text(size = 12),
        strip.text = element_text(size = 12),
        axis.title = element_text(size = 14, face = "bold"),
        legend.text = element_text(size = 12)) +
  geom_text(aes(label = paste0("R² = ",  round(adj_r_squared, 2))), 
            position = position_dodge(width = 1, reverse = TRUE), 
            hjust = 0, show.legend = FALSE,
            data = rsquared %>%  mutate(p_value = 0.01,
                                        term = "Dry season", estimate = 0.05)) +
  annotate("segment", x = 0.6, y = 0.01, xend = 0.6, yend = 0.1,
           arrow = arrow(length = unit(0.5, "cm"), type = "closed"),
           size = 0.9, col = "darkgrey") +
  annotate("segment", x = 0.6, y = -0.01, xend = 0.6, yend = -0.06,
           arrow = arrow(length = unit(0.5, "cm"), type = "closed"),
           size = 0.9, col = "darkgrey") +
  annotate("text", x = 0.75, y = -0.02,
           size = 3, col = "darkgrey", label = "+buffer") +
  annotate("text", x = 0.75, y = 0.06,
           size = 3, col = "darkgrey", label = "-buffer")
